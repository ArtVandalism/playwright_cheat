Playwright Code Snippets

Навигация


# Переход по URL
await page.goto('https://example.com')

# Перезагрузка страницы
await page.reload()

# Вперед/назад по истории
await page.go_back()
await page.go_forward()

Работа с элементами


# Кликнуть элемент
await page.click('button')

# Ввести текст в поле ввода
await page.type('input[name="username"]', 'your_username')

# Нажать клавишу Enter
await page.press('input[name="password"]', 'Enter')

# Получить текстовое содержимое элемента
text = await page.text_content('h1')

# Выбор опции в раскрывающемся списке
await page.select_option('select', label='Option 1')

# Ожидание появления или видимости элементов
await page.wait_for_selector('div#my-element', state='visible')

# Проверка, существует ли элемент
assert await page.locator("button").is_visible()

# Проверка заголовка страницы
assert await page.title() == "Expected Title"

Алерты и диалоговые окна


# Обработка JS-алерта
page.on('dialog', lambda dialog: await dialog.accept())
page.on('dialog', lambda dialog: await dialog.dismiss())

Клавиатура и мышь


# Ввести текст
await page.type('input', 'Hello, Playwright!')

# Нажать и отпустить клавишу
await page.keyboard.press('Enter')
await page.keyboard.release('Shift')

# Передвинуть мышь и кликнуть
await page.mouse.move(100, 100)
await page.mouse.click()

Выполнение (eval) Javascript-кода


# Выполнить JavaScript-код в контексте страницы
result = await page.evaluate('1 + 2')

Фреймы


# Переход к фрейму по имени, id или индексу
frame = await page.frame(name='frameName')
frame = await page.frame(index=0)

# Выполнить код в контексте фрейма
frame = await page.frame(index=0)
await frame.evaluate('console.log("Hello from frame!")')
